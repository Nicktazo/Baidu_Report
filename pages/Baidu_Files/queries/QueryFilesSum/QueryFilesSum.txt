  SELECT
      COUNT(*) AS fileCount,
      SUM(f.[售价_合计]) AS totalPrice,
      SUM(f.[合计成本]) AS totalCost,
      SUM(f.[文件总页数]) AS totalPages
  FROM [Baidu_EasyPrint].[dbo].[FileLevel] f
  INNER JOIN [Baidu_EasyPrint].[dbo].[OrderLevel] o
      ON f.[百度单号] = o.[百度单号]
  WHERE o.[客户] = '贤达'
    AND LEFT(o.[百度单号], 8) BETWEEN '{{moment(StartDate.selectedDate).format("YYYYMMDD")}}'
                                 AND '{{moment(EndDate.selectedDate).format("YYYYMMDD")}}'
