-- 声明变量
DECLARE @StartMonth VARCHAR(6) = '202508';
DECLARE @EndMonth VARCHAR(6) = '202512';
DECLARE @Customer NVARCHAR(50) = N'贤达';

-- 查询订单主表
SELECT 
    ID, 
    LEFT(百度单号, 6) AS 订单年月,
    客户, 百度单号, 省, 市, 快递单号, 百度价格, 印厂, 是否有效, 
    订单状态, 成本合计, 售价合计, 计费重量, 物流成本, 物流售价, 
    包材成本, 预估重量, 物流成本_预估, 物流售价_预估
FROM OrderLevel
WHERE 客户 = @Customer
  AND LEFT(百度单号, 6) BETWEEN @StartMonth AND @EndMonth
ORDER BY 百度单号;
